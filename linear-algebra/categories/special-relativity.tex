
\section{Special relativity}
\begin{axm}
	Linearity
	\begin{align*}
		\lambda
		\begin{pmatrix}
			a_{11} a_{12} \\
			a_{21} a_{22}
		\end{pmatrix}
		\begin{pmatrix}
			z \\
			ct
		\end{pmatrix}
		 & =
		\begin{pmatrix}
			z' \\
			ct'
		\end{pmatrix}
	\end{align*}
\end{axm}
\begin{axm}
	Scaled kinetic translation
	\begin{align*}
		z' = \mu(z - vt)
	\end{align*}
\end{axm}
\begin{axm}
	Invariance
	\begin{align*}
		\lambda^2(z^2 - c^2t^2) & = ({z'}^2 - c^2{t'}^2)
	\end{align*}
\end{axm}
